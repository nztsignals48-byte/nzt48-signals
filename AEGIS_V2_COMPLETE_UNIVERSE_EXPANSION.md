### PHASE 1: CAPITAL PRESERVATION FRAMEWORK

**Purpose**: Calculate maximum position sizes based on Kelly Criterion and ruin probability

**Inputs**:
- Account size: £10,000
- Target daily return: 0.35-0.55% = £35-55
- Max daily loss (circuit breaker): -4.0% = -£400
- Ruin probability threshold: <0.1%

**Algorithm**:
```python
class CapitalPreservation:
    """
    Phase 1: Calculate Kelly Criterion position sizing.
    """

    def __init__(self, account_size=10000, kelly_fraction=0.25):
        self.account_size = account_size
        self.kelly_fraction = kelly_fraction  # Fractional Kelly (0.25-0.5x)

    def calculate_kelly(self, win_rate, avg_win, avg_loss):
        """
        Kelly Criterion: f* = (p×b - q) / b
        where:
        - p = win_rate (0.4-0.6)
        - q = 1 - p
        - b = avg_win / avg_loss (1.5-2.0)

        Example:
        - p = 0.45 (45% win rate)
        - avg_win = 1.5% (0.015)
        - avg_loss = 1.0% (0.010)
        - b = 1.5 / 1.0 = 1.5
        - f* = (0.45 × 1.5 - 0.55) / 1.5 = 0.167 = 16.7% of account
        - Fractional Kelly (0.25x) = 16.7% × 0.25 = 4.17% per trade
        """
        if win_rate < 0.40:
            return 0.0  # Don't trade

        q = 1.0 - win_rate
        b = avg_win / avg_loss if avg_loss > 0 else 1.5

        kelly = (win_rate * b - q) / b

        # Cap at 25% of account (fractional Kelly)
        kelly_fractional = kelly * self.kelly_fraction

        return max(0.0, min(0.25, kelly_fractional))

    def calculate_ruin_probability(self, kelly_pct, num_trades=252):
        """
        Calculate probability of losing entire account.

        Three methods:
        1. Gambler's Ruin formula
        2. Monte Carlo simulation (10,000 trials)
        3. Empirical drawdown analysis
        """
        # Method 1: Gambler's Ruin
        # P(ruin) ≈ (1 - advantage)^num_trades
        advantage = kelly_pct * 2  # Rough approximation
        ruin_prob_1 = (1.0 - advantage) ** num_trades

        # Method 2: Monte Carlo
        ruin_prob_2 = self._monte_carlo_ruin(kelly_pct, num_trades=10000)

        # Method 3: Empirical
        ruin_prob_3 = self._empirical_ruin(kelly_pct)

        # Average of 3 methods
        ruin_probability = (ruin_prob_1 + ruin_prob_2 + ruin_prob_3) / 3

        return ruin_probability

    def _monte_carlo_ruin(self, kelly_pct, num_trials=10000):
        """Simulate ruin over 10,000 random paths."""
        ruin_count = 0

        for _ in range(num_trials):
            balance = self.account_size

            for _ in range(252):  # 252 trading days
                # 45% win, 1.5% avg win, 1.0% avg loss
                is_win = np.random.rand() < 0.45

                if is_win:
                    profit = balance * kelly_pct * 0.015  # 1.5% win
                else:
                    profit = -balance * kelly_pct * 0.010  # -1.0% loss

                balance += profit

                if balance <= 0:
                    ruin_count += 1
                    break

        return ruin_count / num_trials

    def get_position_size(self, kelly_pct, price):
        """Convert Kelly % to share count."""
        max_capital = self.account_size * kelly_pct
        position_size = int(max_capital / price)
        return position_size
```

**Purpose**: Verify ISA compliance every 5 minutes (BINARY gate)

**Checks**:
```python
class ISAAuditor:
    """
    Phase 2: ISA compliance auditor (every 5 minutes).

    ISA Rules:
    - Zero margin debt (BINARY: pass or lose account)
    - ISA-eligible assets only
    - No short positions (except inverse ETPs as hedges)
    - No margin trading
    """

    def verify_compliance(self):
        """
        Full ISA compliance check.
        Returns: (is_compliant, violations_list)
        """
        violations = []

        # Check 1: Margin debt = £0
        margin_debt = self.ibkr.get_margin_debt()
        if margin_debt > 0:
            violations.append(f"CRITICAL: Margin debt £{margin_debt:.2f} (must be £0)")
            self._trigger_emergency_halt()
            return False, violations

        # Check 2: All holdings ISA-eligible
        holdings = self.ibkr.get_holdings()
        for symbol, quantity in holdings.items():
            if not self._is_isa_eligible(symbol):
                violations.append(f"CRITICAL: {symbol} not ISA-eligible")
                self._trigger_emergency_halt()
                return False, violations

        # Check 3: No borrowed shorts (except inverse ETPs)
        for symbol, quantity in holdings.items():
            if quantity < 0 and not self._is_inverse_etp(symbol):
                violations.append(f"CRITICAL: Short position {symbol} {quantity} shares (forbidden)")
                self._trigger_emergency_halt()
                return False, violations

        # Check 4: No margin trading (buying power must be positive)
        buying_power = self.ibkr.get_buying_power()
        if buying_power < 0:
            violations.append(f"CRITICAL: Negative buying power £{buying_power:.2f}")
            self._trigger_emergency_halt()
            return False, violations

        if len(violations) > 0:
            logger.warning(f"ISA violations found: {violations}")
            return False, violations

        return True, []

    def _is_isa_eligible(self, symbol):
        """Check if asset is ISA-eligible."""
        # All Tier 1-3 assets are ISA-eligible
        # FX and commodities (Tier 4C, 4B) are NOT
        isa_ineligible = ['EURUSD', 'GBPUSD', 'CL', 'GC', 'ZW', ...]
        return symbol not in isa_ineligible

    def _is_inverse_etp(self, symbol):
        """Check if position is inverse hedging ETP (allowed short)."""
        inverse_symbols = ['QQQS_INV', 'SPX_INV']
        return symbol in inverse_symbols
```

```python
    Phase 3: Compliance gates (pre-execution).

    Checks BEFORE order submission:
    1. Asset is ISA-eligible
    2. Account will remain at zero margin
    3. Trade doesn't violate concentration limits
    """

    def pre_trade_checks(self, symbol, quantity, current_price):
        """
        Verify trade is ISA-compliant BEFORE execution.
        Returns: (is_compliant, reason)
        """

        # Check 1: ISA eligibility
        if not self._is_isa_eligible(symbol):
            return False, f"{symbol} not ISA-eligible"

        # Check 2: Margin impact
        trade_cost = quantity * current_price
        available_cash = self.ibkr.get_cash_balance()

        if trade_cost > available_cash:
            return False, f"Insufficient cash: need £{trade_cost}, have £{available_cash}"

        # Check 3: Post-trade margin verification
        margin_after = self.ibkr.simulate_margin(symbol, quantity)
        if margin_after > 0:
            return False, f"Would create margin debt £{margin_after}"

        # Check 4: Concentration limit (max 10% per position)
        position_pct = trade_cost / self.account_size
        if position_pct > 0.10:
            return False, f"Position {position_pct:.1%} > 10% limit"

        return True, "All checks passed"
```

### PHASE 4: WHITE REALITY CHECK

**Purpose**: Reject 80% of false positive signals via statistical testing

```python
class WhiteRealityCheck:
    """
    Phase 4: Statistical validation of signals.

    3 Tests:
    1. Bootstrap hypothesis test (Efron)
    2. Deflated Sharpe Ratio (DSR >0.6)
    3. Regime-conditional validation
    """

    def test_signal(self, signal_returns, symbol, regime):
        """
        Test if signal is statistically significant.

        Input: signal_returns = [0.015, -0.010, 0.025, ...]
        Output: (is_significant, DSR, pvalue)
        """

        # Test 1: Deflated Sharpe Ratio
        dsr = self._calculate_dsr(signal_returns)
        if dsr < 0.6:
            return False, dsr, 1.0, "DSR too low"

        # Test 2: Bootstrap hypothesis test
        pvalue = self._bootstrap_test(signal_returns)
        if pvalue > 0.05:
            return False, dsr, pvalue, "Not statistically significant"

        # Test 3: Regime-conditional WR
        regime_wr = self._regime_conditional_wr(symbol, regime)
        if regime_wr < 0.40:
            return False, dsr, pvalue, f"Low WR in {regime}: {regime_wr:.1%}"

        # All tests passed
        return True, dsr, pvalue, "Signal validated"

    def _calculate_dsr(self, returns):
        """
        Deflated Sharpe Ratio (Bailey et al., 2014).
        Adjusts for overfitting and data mining bias.
        """
        sr = np.mean(returns) / np.std(returns)
        skewness = scipy.stats.skew(returns)
        kurtosis = scipy.stats.kurtosis(returns)

        gamma = (skewness * sr) + (kurtosis * sr**2 / 4)
        dsr = sr * np.sqrt((1 - gamma) / (len(returns) - 1))

        return max(0.0, dsr)

    def _bootstrap_test(self, returns, null_mean=0.0, num_samples=1000):
        """Bootstrap hypothesis test (Efron, 1979)."""
        bootstrap_means = [
            np.mean(np.random.choice(returns, len(returns), replace=True))
            for _ in range(num_samples)
        ]
        pvalue = np.sum(np.array(bootstrap_means) <= null_mean) / num_samples
        return pvalue
```

### PHASE 10: EXECUTION QUALITY

```python
class ExecutionQuality:
    """
    Phase 10: Track execution quality metrics.

    Metrics:
    - Slippage: actual price vs model
    - Entry timing: early, mid, late in trend
    - Exit timing: optimal exit price
    """

    def calculate_slippage(self, symbol, intended_price, actual_price):
        """
        Calculate slippage percentage.

        Example:
        - Intended: £25.50
        - Actual: £25.60 (filled higher)
        - Slippage: (25.60 - 25.50) / 25.50 = 0.39%
        """
        slippage = (actual_price - intended_price) / intended_price
        return abs(slippage)

    def calculate_entry_timing_score(self, entry_price, daily_low, daily_high):
        """
        Score entry timing on 0-1 scale.

        0.0 = worst (bought at daily high)
        0.5 = neutral (bought at daily middle)
        1.0 = best (bought at daily low)
        """
        range_size = daily_high - daily_low
        if range_size == 0:
            return 0.5

        distance_from_low = entry_price - daily_low
        timing_score = 1.0 - (distance_from_low / range_size)

        return max(0.0, min(1.0, timing_score))
```

```python
        - signal: {'underlying': 'NVDA', 'direction': 'BUY', 'confidence': 7.8}
        - position_size_output: {'size': 500, 'symbol': '', ...}

        Output:
        - execution_result: {'order_id', 'symbol', 'fill_price', ...}

        if not is_isa_eligible:
            return {'status': 'REJECTED', 'reason': 'Not ISA-eligible'}

        # Step 2: Zero margin check
        if self.ibkr.get_margin_debt() > 0:
            return {'status': 'REJECTED', 'reason': 'Margin debt > 0'}

        # STEP 3: GET SYMBOL (LEVERAGE PRIORITIZATION)
        underlying = signal['underlying']
        confidence = signal['confidence']
        lse_open = self._is_lse_open()

        # Select symbol based on leverage priority
        if underlying in self.UNDERLYING_TO_ETP:
            etp_config = self.UNDERLYING_TO_ETP[underlying]

            if confidence > 8.0 and etp_config.get('high_conf'):
                symbol = etp_config['high_conf']
                leverage = 5.0
            # Priority 2: LSE open + 3x available
            elif lse_open and etp_config.get('lse_open'):
                symbol = etp_config['lse_open']
                leverage = 3.0
            # Priority 3: LSE closed or no ETP → direct
            else:
                symbol = etp_config.get('lse_closed', underlying)
                leverage = 1.0
        else:
            symbol = underlying
            leverage = 1.0

        # STEP 4: SUBMIT ORDER
        size = int(position_size_output['size'])

        try:
            order_result = self.ibkr.place_order(
                symbol=symbol,
                quantity=size,
                direction=signal['direction'],
                order_type='MARKET',
                time_in_force='DAY',
            )

            # STEP 5: LOG EXECUTION
            self.log_execution({
                'timestamp': time.time(),
                'signal_underlying': underlying,
                'executed_symbol': symbol,
                'leverage': leverage,
                'size': size,
                'fill_price': order_result['fill_price'],
                'order_id': order_result['order_id'],
            })

                'symbol': symbol,
                'fill_price': order_result['fill_price'],
```
